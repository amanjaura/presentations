\section{Tech Labs Objectives}
% --------------------- Objectives --------------------------
\frame{
	\frametitle{Primary Objectives}
	\begin{enumerate}
		\item Understanding how Odoo (a 3-tier Web Application) works
		\pause
		\item Gain some kind of familiarity with installing and configuring software on Linux
		\pause
		\item Know the applications most use in business, and what they're good at
		\pause
		\item Become familiar working collaboratively as a team
		\pause
		\item Explore areas of interest in a safe and helpful space
	\end{enumerate}
}

\frame{
	\frametitle{Secondary Objectives}
	\begin{enumerate}
		\item Install a Linux desktop
		\pause
		\item Learn some basic commands (CLI) and apps (GUI) to be productive in Linux
		\pause
		\item Install and configure a web server
		\pause
		\item Install and configure a database
		\pause
		\item Understanding basic concepts about how the Internet works
	\end{enumerate}
}
